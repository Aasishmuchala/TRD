    # God's Eye Backend — Environment Variables
    # Copy this file to .env and fill in your values.
    # In Railway: set these in the Railway service's "Variables" tab.

    # ─── LLM Provider ────────────────────────────────────────────────────────────
    # Which LLM provider to use. Options: "openai" | "nous" | "custom"
    GODS_EYE_LLM_PROVIDER=openai

    # Your LLM API key. For OpenAI: starts with sk-
    # Railway: add this as a secret variable (not shown in logs)
    LLM_API_KEY=sk-your-openai-key-here

    # Alternative: OPENAI_API_KEY is also read if LLM_API_KEY is not set
    # OPENAI_API_KEY=sk-your-openai-key-here

    # Override the LLM inference base URL (optional — for custom endpoints or proxies)
    # LLM_INFERENCE_URL=https://your-custom-endpoint.com/v1

    # Model name to use. Examples: o4-mini, gpt-4o, gpt-4o-mini
    GODS_EYE_MODEL=o4-mini

    # Set to true to use mock responses (no LLM calls). Useful for testing.
    # Production should always be false.
    GODS_EYE_MOCK=false

    # ─── Database ────────────────────────────────────────────────────────────────
    # Path to the SQLite database file.
    # Railway: set this to /app/data/gods_eye.db and mount a persistent volume at /app/data
    GODS_EYE_DB_PATH=/app/data/gods_eye.db

    # ─── Skills / Learning ───────────────────────────────────────────────────────
    # Directory where the SkillStore writes extracted agent skills.
    # Railway: set to /app/skills and mount a persistent volume at /app/skills
    GODS_EYE_LEARNING_SKILL_DIR=/app/skills

    # Enable automatic skill extraction from completed simulations (default: true)
    GODS_EYE_LEARNING=true

    # ─── CORS ────────────────────────────────────────────────────────────────────
    # Comma-separated list of allowed frontend origins.
    # PRODUCTION: replace with your actual Vercel deployment URL.
    # Example: GODS_EYE_CORS_ORIGINS=https://gods-eye.vercel.app
    # Multiple origins: GODS_EYE_CORS_ORIGINS=https://gods-eye.vercel.app,https://gods-eye-preview.vercel.app
    # DO NOT use wildcard (*) — credentials (Authorization header) require exact origins.
    GODS_EYE_CORS_ORIGINS=https://your-frontend.vercel.app

    # ─── Server ──────────────────────────────────────────────────────────────────
    # Environment name. Use "production" on Railway.
    GODS_EYE_ENV=production

    # Log level. Options: debug | info | warning | error | critical
    GODS_EYE_LOG_LEVEL=info

    # ─── Legacy (backward compatibility) ─────────────────────────────────────────
    # If you have a Claude API key from an older setup, it maps to LLM_API_KEY.
    # Prefer LLM_API_KEY for new deployments.
    # CLAUDE_API_KEY=sk-ant-your-key-here
    ```

    # Copy this file to .env (for local dev) or set in Vercel Dashboard → Settings → Environment Variables.
    #
    # Vite env vars MUST start with VITE_ to be available in browser code.
    # Variables without VITE_ prefix are server-side only and NOT accessible in the browser.

    # ─── API Base URL ────────────────────────────────────────────────────────────
    # Base path for all REST API calls.
    # Local dev: leave unset — Vite proxy routes /api/* to localhost:8000
    # Production (Vercel): set to your Railway backend URL
    # Example: VITE_API_BASE=https://gods-eye-backend.up.railway.app/api
    VITE_API_BASE=https://your-railway-backend.up.railway.app/api

    # ─── WebSocket Base URL ──────────────────────────────────────────────────────
    # Base URL for the WebSocket simulation stream.
    # Local dev: leave unset — falls back to same-host (Vite proxy handles it)
    # Production (Vercel): MUST be set to the Railway backend WebSocket URL
    # Format: wss://hostname (no trailing slash, no path)
    # Example: VITE_WS_BASE=wss://gods-eye-backend.up.railway.app
    VITE_WS_BASE=wss://your-railway-backend.up.railway.app
    ```
