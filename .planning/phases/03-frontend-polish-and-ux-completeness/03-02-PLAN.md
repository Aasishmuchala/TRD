```python
async def _fetch_bank_nifty(self) -> Dict[str, Any]:
    """Fetch Bank Nifty index data from NSE."""
    cached = cache.get("bank_nifty")
    if cached:
        return cached

    try:
        async with httpx.AsyncClient(
            headers=self._nse_headers(),
            timeout=10.0,
            follow_redirects=True,
        ) as client:
            # First hit NSE homepage to get cookies (same pattern as _fetch_nifty50)
            await client.get("https://www.nseindia.com", timeout=5.0)
            resp = await client.get(
                "https://www.nseindia.com/api/equity-stockIndices?index=NIFTY%20BANK",
                timeout=10.0,
            )
            resp.raise_for_status()
            payload = resp.json()
            # NSE returns {"data": [{"indexSymbol": "NIFTY BANK", "last": ..., ...}]}
            items = payload.get("data", [])
            index_data = next(
                (x for x in items if "BANK" in x.get("indexSymbol", "").upper()),
                {}
            )
            if not index_data:
                raise ValueError("Bank Nifty data not found in response")

            last = float(index_data.get("last", 0) or 0)
            prev = float(index_data.get("previousClose", 0) or index_data.get("yearLow", 0) or 0)
            change = last - prev
            change_pct = (change / prev * 100) if prev > 0 else 0.0

            result = {
                "last": last,
                "open": float(index_data.get("open", 0) or 0),
                "high": float(index_data.get("high", 0) or 0),
                "low": float(index_data.get("low", 0) or 0),
                "prev_close": prev,
                "change": change,
                "change_pct": change_pct,
            }
            cache.set("bank_nifty", result, ttl=MarketCache.DEFAULT_TTLS.get("spot", 60))
            return result

    except Exception:
        # Graceful fallback — return zeroed values
        return {"last": 0, "open": 0, "high": 0, "low": 0, "prev_close": 0, "change": 0, "change_pct": 0}
```
